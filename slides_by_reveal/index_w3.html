function cellClick(cell){
    machine = Math.floor(Math.random() * 3);
    var csel = choices[machine];
    var hsel = cell.getAttribute("data-human-choice");
    console.log( "machine selected: " + csel 
                 + " | you selected: " + hsel );
    if ( csel == hsel ) 
        message = " IT'S A TIE! "; 
    else
        looseTo[csel] == hsel 
            ? message = " YOU WIN! " 
            : message = " MACHINE WINS! ";  
    console.log(message);
}
